import numpy as np 
def heuristics_v1(Positions, Best_pos, Best_score, rg):
    SearchAgents_no = Positions.shape[0]
    dim = Positions.shape[1]

    lb_array = np.zeros((SearchAgents_no, dim))
    ub_array = np.ones((SearchAgents_no, dim))

    rand_adjust = lb_array + (ub_array - lb_array) * np.random.rand(*Positions.shape)
    Positions = np.where((Positions < lb_array) | (Positions > ub_array), rand_adjust, Positions)

    #EVOLVE-START
    # Opposition-based learning
    opposite_pos = lb_array + ub_array - Positions
    combined_pos = np.vstack((Positions, opposite_pos))
    
    # Nonlinear convergence factor
    a = 2 - 2*(np.log(np.linspace(1, np.exp(1), SearchAgents_no)))
    r1 = np.random.rand(SearchAgents_no, dim)
    r2 = np.random.rand(SearchAgents_no, dim)
    
    # Adaptive position update
    A = 2*a*r1 - a
    C = 2*r2
    D = np.abs(C*Best_pos - combined_pos[:SearchAgents_no])
    Positions = Best_pos - A*D
    #EVOLVE-END       

    return Positions
