import numpy as np 
def heuristics_v2(Positions, Best_pos, Best_score, rg):
    SearchAgents_no = Positions.shape[0]
    dim = Positions.shape[1]

    lb_array = np.zeros((SearchAgents_no, dim))
    ub_array = np.ones((SearchAgents_no, dim))

    rand_adjust = lb_array + (ub_array - lb_array) * np.random.rand(*Positions.shape)
    Positions = np.where((Positions < lb_array) | (Positions > ub_array), rand_adjust, Positions)

    #EVOLVE-START
    # Adaptive neighborhood search with safe sampling
    min_size = 3
    max_possible = max(1, SearchAgents_no - 1)  # Ensure at least 1
    neighborhood_size = min(max(min_size, int(SearchAgents_no * rg)), max_possible)
    
    if SearchAgents_no >= min_size:
        neighbor_indices = np.array([np.random.choice(SearchAgents_no, size=neighborhood_size, replace=False) for _ in range(SearchAgents_no)])
    else:
        neighbor_indices = np.tile(np.arange(SearchAgents_no), (SearchAgents_no, 1))
    
    # Memory-guided exploration
    memory_ratio = 0.3 * (1 - rg)
    memory_mask = np.random.rand(SearchAgents_no, dim) < memory_ratio
    memory_guide = np.random.rand(SearchAgents_no, dim) * Best_pos
    
    # Position update
    local_bests = Positions[neighbor_indices].mean(axis=1)
    Positions = (1-memory_mask)*((1-rg)*local_bests + rg*Positions) + memory_mask*memory_guide
    #EVOLVE-END       

    return Positions
