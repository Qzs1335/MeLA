import numpy as np 
def heuristics_v2(Positions, Best_pos, Best_score, rg):
    SearchAgents_no = Positions.shape[0]
    dim = Positions.shape[1]

    lb_array = np.zeros((SearchAgents_no, dim))
    ub_array = np.ones((SearchAgents_no, dim))

    rand_adjust = lb_array + (ub_array - lb_array) * np.random.rand(*Positions.shape)
    Positions = np.where((Positions < lb_array) | (Positions > ub_array), rand_adjust, Positions)

    #EVOLVE-START
    F = 0.5 * (1 + rg)  # Adaptive scaling factor
    CR = 0.9 - 0.5*rg   # Adaptive crossover rate
    
    # Differential evolution mutation with fixed size handling
    idxs = np.random.permutation(SearchAgents_no)
    split1 = SearchAgents_no // 3
    split2 = 2 * split1
    # Ensure we have three distinct groups of equal size
    a = Positions[idxs[:split1]]
    b = Positions[idxs[split1:split2]]
    c = Positions[idxs[split2:split2+split1]]  # Make sure c has same size as a and b
    
    # Broadcast the mutation operation properly
    mutant = a + F*(b - c)
    
    # Binomial crossover with matching dimensions
    cross_points = np.random.rand(SearchAgents_no, dim) < CR
    # Resize mutant to match original Positions shape
    mutant_full = np.tile(mutant, (1, 1, 1))[:SearchAgents_no, :dim]
    Positions = np.where(cross_points, mutant_full, Positions)
    #EVOLVE-END
    
    return Positions
