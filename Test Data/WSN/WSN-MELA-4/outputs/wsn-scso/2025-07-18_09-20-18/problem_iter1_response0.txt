import numpy as np 
def heuristics_v1(Positions, Best_pos, Best_score, rg):
    SearchAgents_no = Positions.shape[0]
    dim = Positions.shape[1]

    lb_array = np.zeros((SearchAgents_no, dim))
    ub_array = np.ones((SearchAgents_no, dim))

    rand_adjust = lb_array + (ub_array - lb_array) * np.random.rand(*Positions.shape)
    Positions = np.where((Positions < lb_array) | (Positions > ub_array), rand_adjust, Positions)

    #EVOLVE-START
    F = 0.5 * (1 + rg)  # Adaptive scaling factor
    CR = 0.9 - 0.5*rg   # Adaptive crossover rate
    
    # Differential evolution mutation
    idxs = np.random.permutation(SearchAgents_no)
    a, b, c = Positions[idxs[:SearchAgents_no//3]], Positions[idxs[SearchAgents_no//3:2*SearchAgents_no//3]], Positions[idxs[2*SearchAgents_no//3:]]
    mutant = a + F*(b - c)
    
    # Binomial crossover
    cross_points = np.random.rand(*Positions.shape) < CR
    Positions = np.where(cross_points, mutant, Positions)
    #EVOLVE-END
    
    return Positions
