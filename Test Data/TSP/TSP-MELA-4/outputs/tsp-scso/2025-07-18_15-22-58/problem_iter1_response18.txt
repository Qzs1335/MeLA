2. Adding logarithmic scaling can better handle varying distance magnitudes
3. Incorporating minimum distance normalization improves relative weighting
4. Small epsilon prevents division by zero while maintaining gradient}

import numpy as np
def heuristics_v1(distance_matrix):
    #EVOLVE-START
    epsilon = 1e-10
    min_dist = np.min(distance_matrix[distance_matrix > 0])
    normalized = distance_matrix / min_dist
    return np.log(1 + (1 / (normalized + epsilon)))
    #EVOLVE-END
    return 1 / distance_matrix
