2. We can improve by combining logarithmic transformation for better sensitivity to small distances
3. Maintain epsilon to prevent division by zero
4. Use softmax to normalize outputs while preserving relative differences}

import numpy as np 
def heuristics_v2(distance_matrix):
    #EVOLVE-START
    epsilon = 1e-15
    log_transformed = -np.log(distance_matrix + epsilon)
    return np.exp(log_transformed) / np.sum(np.exp(log_transformed))
    #EVOLVE-END       
    return 1 / distance_matrix
