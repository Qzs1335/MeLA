import numpy as np 
def heuristics_v1(distance_matrix):
    #EVOLVE-START
    epsilon = 1e-10
    return np.sqrt(np.log(1 / (distance_matrix + epsilon)))
    #EVOLVE-END       
    return 1 / distance_matrix
