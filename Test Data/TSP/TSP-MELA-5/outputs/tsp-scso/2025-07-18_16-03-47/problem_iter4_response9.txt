import numpy as np 
def heuristics_v2(distance_matrix):
    #EVOLVE-START
    epsilon = 1e-15
    log_dist = np.log(distance_matrix + epsilon)
    return 1 / (distance_matrix + epsilon) + np.exp(-np.abs(log_dist))
    #EVOLVE-END       
    return 1 / distance_matrix
