import numpy as np
def heuristics_v2(node_attr, node_constraint):
    #EVOLVE-START
    n = node_attr.shape[0]
    if n == 0:
        return np.zeros((0, 0))
    
    heuristic = np.zeros((n, n))
    remaining_cap = node_constraint - np.cumsum(node_attr)
    
    for i in range(n):
        current_cap = remaining_cap[i]
        for j in range(n):
            if node_attr[j] <= current_cap:
                heuristic[i,j] = np.exp(-node_attr[j]/current_cap)
    return heuristic
