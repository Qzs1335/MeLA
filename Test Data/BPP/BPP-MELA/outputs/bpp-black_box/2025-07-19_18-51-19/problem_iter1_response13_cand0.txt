import numpy as np
def heuristics_v2(node_attr, node_constraint):
    #EVOLVE-START
    weights = np.asarray(node_attr).reshape(-1, 1)  # Ensure column vector
    capacity = max(node_constraint, 1e-6)  # Avoid division by zero
    
    # Calculate inverse weight-distance matrix
    dist_matrix = np.abs(weights - weights.T)
    heuristic = (capacity - dist_matrix) / capacity
    np.fill_diagonal(heuristic, 0)  # Avoid self-selection
    return heuristic
    #EVOLVE-END
