import numpy as np
def heuristics_v2(node_attr, node_constraint):
    #EVOLVE-START
    sizes = node_attr if node_attr.ndim == 1 else node_attr[:, 0]
    dist = np.abs(sizes - sizes[:, None])
    heuristic = np.exp(-dist/node_constraint)
    np.fill_diagonal(heuristic, 0)
