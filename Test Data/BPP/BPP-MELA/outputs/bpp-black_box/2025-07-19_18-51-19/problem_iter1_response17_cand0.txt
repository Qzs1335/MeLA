def heuristics_v2(node_attr, node_constraint):

    n = node_attr.shape[0]
    if n == 0:
        return np.zeros((0, 0))
    
    heuristic = np.zeros((n, n))
    for i in range(n):
        if i == 0:
            remaining_cap = node_constraint
        else:
            remaining_cap = node_constraint - np.sum(node_attr[:i])
