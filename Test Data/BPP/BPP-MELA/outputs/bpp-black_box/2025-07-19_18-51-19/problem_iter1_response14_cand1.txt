import numpy as np
def heuristics_v2(node_attr, node_constraint):
    #EVOLVE-START
    if node_attr.ndim == 1:
        weights = node_attr/node_constraint
    else:
        weights = node_attr[:,0]/node_constraint
    heuristic = np.outer(weights, 1-weights)
    np.fill_diagonal(heuristic, 0)
