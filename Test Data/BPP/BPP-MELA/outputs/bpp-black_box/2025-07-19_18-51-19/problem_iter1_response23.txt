import numpy as np 
def heuristics_v1(node_attr, node_constraint):
    #EVOLVE-START
    n = node_attr.shape[0]
    sizes = node_attr[:,0]
    heur = np.zeros((n,n))
    for i in range(n):
        for j in range(n):
            if i != j:
                size_diff = abs(sizes[i]-sizes[j])

    np.fill_diagonal(heur, 0)
