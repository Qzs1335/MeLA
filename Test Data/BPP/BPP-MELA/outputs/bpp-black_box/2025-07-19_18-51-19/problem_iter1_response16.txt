import numpy as np
def heuristics_v1(node_attr, node_constraint):
    #EVOLVE-START
    sizes = node_attr[:, 0]
    n = len(sizes)
    heuristic = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                heuristic[i,j] = 1 / (abs(sizes[i] + sizes[j] - node_constraint) + 1e-6)
    return heuristic / heuristic.max()
    #EVOLVE-END
