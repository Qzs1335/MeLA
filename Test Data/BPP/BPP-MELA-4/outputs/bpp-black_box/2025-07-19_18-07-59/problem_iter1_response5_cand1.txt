import numpy as np
def heuristics_v2(node_attr, node_constraint):
    #EVOLVE-START
    # Ensure inputs are numpy arrays
    node_attr = np.asarray(node_attr)
    node_constraint = np.asarray(node_constraint)
    
    # Handle 1D or 2D node_attr
    if node_attr.ndim == 1:
        sizes = node_attr.reshape(-1,1)
    else:
        sizes = node_attr[:,0].reshape(-1,1)
    
    # Ensure capacities is properly shaped
    capacities = np.asarray(node_constraint).reshape(-1,1)
    
    # Calculate size ratios with protection against division by zero
    min_vals = np.minimum(sizes, capacities)
    max_vals = np.maximum(sizes, capacities)
    size_ratios = np.divide(min_vals, max_vals, where=max_vals!=0)
    
    return np.exp(-5*(1-size_ratios))
