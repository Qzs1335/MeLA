import numpy as np 
def heuristics_v1(node_attr, node_constraint):
    #EVOLVE-START
    n = node_attr.shape[0]
    sizes = node_attr[:,0].reshape(-1,1)
    heuristic = 1/(1 + np.abs(sizes - sizes.T))  # Size similarity

    np.fill_diagonal(heuristic, 0)  # Prevent self-assignment
