import numpy as np
def heuristics_v1(node_attr, node_constraint):
    #EVOLVE-START
    n = node_attr.shape[0]
    dists = np.zeros((n,n))
    for i in range(n):
        for j in range(n):
            dists[i,j] = np.linalg.norm(node_attr[i]-node_attr[j])
