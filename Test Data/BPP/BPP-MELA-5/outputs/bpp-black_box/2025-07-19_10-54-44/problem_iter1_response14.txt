import numpy as np
def heuristics_v1(node_attr, node_constraint):
    #EVOLVE-START
    n = node_attr.shape[0]
    sizes = node_attr[:, 0]
    heuristic = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                remaining = node_constraint - sizes[i] - sizes[j]
                heuristic[i,j] = 1/(abs(sizes[i] - sizes[j]) + 0.1) + remaining/node_constraint
    return heuristic
    #EVOLVE-END
