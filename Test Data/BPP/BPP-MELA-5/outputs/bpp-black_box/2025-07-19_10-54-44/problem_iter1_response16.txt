1.Current heuristic uses uniform weights (ones matrix) which provides no guidance
2.Should incorporate item sizes and bin capacity
3.Larger items should have lower transition probabilities to smaller items
4.Normalize weights by remaining capacity
5.Add small random noise to prevent premature convergence}

import numpy as np 
def heuristics_v1(node_attr, node_constraint):
    #EVOLVE-START
    n = node_attr.shape[0]
    sizes = node_attr[:,0]
    h = np.zeros((n,n))
    for i in range(n):
        remaining = node_constraint - sizes[i]
        for j in range(n):
            if i != j and sizes[j] <= remaining:
                h[i,j] = 1/(1 + sizes[j]/remaining)
    h += 0.01*np.random.rand(n,n)  # exploration factor
    np.fill_diagonal(h, 0)
    return h
    #EVOLVE-END
