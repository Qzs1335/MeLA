import numpy as np
def heuristics_v1(node_attr, node_constraint):
    #EVOLVE-START
    n = node_attr.shape[0]
    sizes = node_attr[:, 0]
    h = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                h[i,j] = 1/(abs(sizes[i] + sizes[j] - node_constraint) + 1e-6)
    np.fill_diagonal(h, 0)
    return h
    #EVOLVE-END
