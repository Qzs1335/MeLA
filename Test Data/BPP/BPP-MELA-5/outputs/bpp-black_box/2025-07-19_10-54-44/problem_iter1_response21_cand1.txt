import numpy as np
def heuristics_v2(node_attr, node_constraint):
    #EVOLVE-START
    node_attr = np.asarray(node_attr).reshape(-1, 1) if len(node_attr.shape) == 1 else node_attr
    n = node_attr.shape[0]
    sizes = node_attr[:,0] if node_attr.shape[1] > 1 else node_attr.flatten()
    h = np.zeros((n,n))
    for i in range(n):
        for j in range(n):
            if i != j:
                denominator = node_constraint - sizes[i]
                h[i,j] = (sizes[i] + sizes[j]) / denominator if denominator != 0 else 0
    return h
