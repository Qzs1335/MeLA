import numpy as np
def heuristics_v2(node_attr, node_constraint):
    #EVOLVE-START
    if node_attr.ndim == 1:
        weights = node_attr
    else:
        weights = node_attr[:,0]
    heuristic = np.abs(np.subtract.outer(weights, weights)) / node_constraint
    return 1 / (1 + heuristic)
