import numpy as np
def heuristics_v2(node_attr, node_constraint):
    #EVOLVE-START
    weights = node_attr if node_attr.ndim == 1 else node_attr[:,0]
    weights = weights.reshape(-1, 1)  # Ensure 2D for broadcasting
    dist_matrix = np.abs(weights - weights.T) + 1e-10  # Add small value to avoid division by zero
    capacity_ratio = np.minimum(weights, weights.T) / node_constraint
    return 0.7 * (1/dist_matrix) + 0.3 * capacity_ratio
