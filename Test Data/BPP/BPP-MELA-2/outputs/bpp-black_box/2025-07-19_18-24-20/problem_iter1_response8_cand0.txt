import numpy as np
def heuristics_v2(node_attr, node_constraint):
    #EVOLVE-START
    # Ensure node_attr is 2D (n,1) array
    if len(node_attr.shape) == 1:
        node_attr = node_attr[:, np.newaxis]
    
    n = node_attr.shape[0]
    sizes = node_attr[:, 0]
    heuristic = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if sizes[i] + sizes[j] <= node_constraint:
                heuristic[i,j] = 1/(1 + abs(sizes[i] - sizes[j]))
    return heuristic
