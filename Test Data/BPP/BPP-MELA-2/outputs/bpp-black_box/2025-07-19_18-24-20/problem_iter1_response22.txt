import numpy as np
def heuristics_v1(node_attr, node_constraint):
    #EVOLVE-START
    n = node_attr.shape[0]
    weights = node_attr[:,0]
    capacity = node_constraint[0]
    heuristic = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                remaining = capacity - weights[i] - weights[j]
                heuristic[i,j] = 1/(1 + abs(remaining)) + 0.1*np.random.rand()
    np.fill_diagonal(heuristic, 0)
    return heuristic/heuristic.max()
    #EVOLVE-END
