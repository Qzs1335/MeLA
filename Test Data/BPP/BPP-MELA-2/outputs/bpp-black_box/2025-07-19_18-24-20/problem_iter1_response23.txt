import numpy as np
def heuristics_v1(node_attr, node_constraint):
    #EVOLVE-START
    n = node_attr.shape[0]
    dist = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                dist[i,j] = 1/(1 + np.linalg.norm(node_attr[i]-node_attr[j]))
            else:
                dist[i,j] = 1
    return dist * (node_constraint[:,None] + node_constraint)/node_constraint.max()
