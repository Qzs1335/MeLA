import numpy as np
def heuristics_v1(node_attr, node_constraint):
    #EVOLVE-START
    n = node_attr.shape[0]
    sizes = node_attr[:, 0]
    heuristic = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if sizes[i] + sizes[j] <= node_constraint:
                heuristic[i,j] = 1/(1 + abs(sizes[i] - sizes[j]))
    return heuristic
