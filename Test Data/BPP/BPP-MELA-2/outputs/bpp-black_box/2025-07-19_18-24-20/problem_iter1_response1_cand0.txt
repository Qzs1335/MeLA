import numpy as np 
def heuristics_v2(node_attr, node_constraint):
    #EVOLVE-START
    if len(node_attr.shape) == 1:
        weights = node_attr
    else:
        weights = node_attr[:, 0] if node_attr.shape[1] > 0 else node_attr.flatten()
    
    dist_matrix = np.abs(weights.reshape(-1, 1) - weights)
    constraint_matrix = np.minimum(node_constraint.reshape(-1, 1), node_constraint)
    max_constraint = node_constraint.max() if node_constraint.size > 0 else 1
    heuristic = (1 / (1 + dist_matrix)) * (constraint_matrix / max_constraint)
    return heuristic
    #EVOLVE-END
