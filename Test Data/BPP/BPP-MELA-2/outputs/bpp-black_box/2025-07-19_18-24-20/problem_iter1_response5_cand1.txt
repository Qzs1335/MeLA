import numpy as np 
def heuristics_v2(node_attr, node_constraint):
    #EVOLVE-START
    node_attr = np.array(node_attr)
    if len(node_attr.shape) == 1:
        node_attr = node_attr.reshape(-1, 1)
    n = node_attr.shape[0]
    sizes = node_attr[:,0]
    normalized_sizes = sizes / node_constraint
    heuristic = np.outer(normalized_sizes, normalized_sizes)
    return heuristic * (1 + np.eye(n))  # Boost diagonal
