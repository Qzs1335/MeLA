import numpy as np
def heuristics_v1(node_attr, node_constraint):
    #EVOLVE-START
    n = node_attr.shape[0]
    weights = node_attr[:,0]
    heuristic = np.zeros((n,n))
    for i in range(n):
        for j in range(n):
            if i != j:
                remaining = node_constraint - weights[i]
                heuristic[i,j] = 1/(1 + abs(weights[j] - remaining))
    return heuristic
    #EVOLVE-END
