import numpy as np 
def heuristics_v2(node_attr, node_constraint):
    #EVOLVE-START
    n = len(node_attr)
    weights = node_attr
    h = np.zeros((n,n))
    for i in range(n):
        for j in range(n):
            if i == j:
                h[i,j] = weights[i]/node_constraint
            else:
                h[i,j] = 1 - abs(weights[i]-weights[j])/node_constraint
    return h
    #EVOLVE-END
